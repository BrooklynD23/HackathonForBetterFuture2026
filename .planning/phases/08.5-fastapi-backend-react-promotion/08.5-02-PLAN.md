    ```typescript
    // === Types matching Python backend responses ===

    export interface Specialist {
      name: string;
      board_role: string;
      metro_region: string;
      company: string;
      title: string;
      expertise_tags: string;
      initials: string;
    }

    export interface CppEvent {
      "Event / Program": string;
      [key: string]: string;  // CSV columns vary
    }

    export interface PipelineRecord {
      event_name: string;
      speaker_name: string;
      match_score: string;
      rank: string;
      stage: string;
      stage_order: string;
    }

    export interface CalendarEvent {
      [key: string]: string;
    }

    export interface PocContact {
      [key: string]: unknown;
    }

    export interface RankedMatch {
      name: string;
      score: number;
      rank: number;
      [key: string]: unknown;
    }

    export interface MatchScore {
      total_score: number;
      factor_scores: Record<string, number>;
      weighted_factor_scores: Record<string, number>;
    }

    // === Helper ===

    async function apiFetch<T>(url: string, options?: RequestInit): Promise<T> {
      const response = await fetch(url, {
        headers: { "Content-Type": "application/json", ...options?.headers },
        ...options,
      });
      if (!response.ok) {
        const body = await response.text();
        throw new Error(`API error ${response.status}: ${body}`);
      }
      return response.json();
    }

    // === Data endpoints ===

    export const fetchSpecialists = () =>
      apiFetch<Specialist[]>("/api/data/specialists");

    export const fetchEvents = () =>
      apiFetch<CppEvent[]>("/api/data/events");

    export const fetchPipeline = () =>
      apiFetch<PipelineRecord[]>("/api/data/pipeline");

    export const fetchCalendar = () =>
      apiFetch<CalendarEvent[]>("/api/data/calendar");

    export const fetchContacts = () =>
      apiFetch<PocContact[]>("/api/data/contacts");

    // === Matching endpoints ===

    export const rankSpeakers = (eventName: string, limit = 5) =>
      apiFetch<RankedMatch[]>("/api/matching/rank", {
        method: "POST",
        body: JSON.stringify({ event_name: eventName, limit }),
      });

    export const scoreSpeaker = (speakerName: string, eventName: string) =>
      apiFetch<MatchScore>("/api/matching/score", {
        method: "POST",
        body: JSON.stringify({ speaker_name: speakerName, event_name: eventName }),
      });

    // === Outreach endpoints ===

    export const generateEmail = (speakerName: string, eventName: string) =>
      apiFetch<{ email: string }>("/api/outreach/email", {
        method: "POST",
        body: JSON.stringify({ speaker_name: speakerName, event_name: eventName }),
      });

    export const generateIcs = (
      eventName: string,
      eventDate?: string,
      location?: string,
      description?: string
    ) =>
      apiFetch<{ ics_content: string }>("/api/outreach/ics", {
        method: "POST",
        body: JSON.stringify({
          event_name: eventName,
          event_date: eventDate,
          location,
          description,
        }),
      });
    ```
